```jsx
<>

    Start
  </Text>
  <Text as="p" align="center">
    Center
  </Text>
  <Text as="p" align="end">
    End
  </Text>
  <Text as="p" align="left">
    Left
  </Text>
  <Text as="p" align="right">
    Right
  </Text>
</>
```
