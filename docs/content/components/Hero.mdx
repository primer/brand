import {Label} from '@primer/react'

```js
import {Hero} from '@primer/react-brand'
```

## Examples

### Default

```jsx live
<Hero
  heading="This is my super sweet hero heading"
  description="Lorem ipsum dolor sit amet, consectetur adipiscing elit. In sapien sit ullamcorper id. Aliquam luctus sed turpis felis nam pulvinar risus elementum."
  primaryAction={{
    text: 'Primary action',
    href: '#',
  }}
  secondaryAction={{
    text: 'Secondary action',
    href: '#',
  }}
  align="center"
/>
```

### Alignment

`Hero` text alignment can be adjusted by changing the `align` prop.

```jsx live
<Hero
  heading={
    <>
      Blazing fast cloud
      <br />
      developer environments
    </>
  }
  description="Visual Studio Code backed by high performance VMs that start in seconds."
  primaryAction={{
    text: 'Primary action',
    href: '#',
  }}
/>
```

## Component props

### Hero

| Name                                    | Type                                                                                                               | Default  | Description                             |
| :-------------------------------------- | :----------------------------------------------------------------------------------------------------------------- | :------: | :-------------------------------------- |
| `heading` <Label>Required</Label>       | `string`                                                                                                           |          | Primary heading, which defaults to a H1 |
| `primaryAction` <Label>Required</Label> | [Object](https://github.com/primer/react-brand/blob/ca2d3002ba6f7ca9c719f80077c4d95246f33410/src/Hero/Hero.tsx#L8) |          | Primary call to action                  |
| `secondaryAction`                       | [Object](https://github.com/primer/react-brand/blob/ca2d3002ba6f7ca9c719f80077c4d95246f33410/src/Hero/Hero.tsx#L8) |          | Optional secondary call to action       |
| `align`                                 | `'center'`, `'left'`                                                                                               | `'left'` | Optional text alignment                 |
| `description`                           | `string`                                                                                                           |          | Optional accompanying body text         |
