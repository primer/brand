import {CTAForm, InlineLink} from '@primer/react-brand'

```js
import {CTAForm} from '@primer/react-brand'
```

## Examples

### Default

```jsx live
<CTAForm
  buttonLabel="Subscribe"
  checkboxLabel={
    <>
      I agree to the{' '}
      <InlineLink href="www.github.com">terms and conditions</InlineLink> and{' '}
      <InlineLink href="www.github.com">privacy policy</InlineLink>
    </>
  }
  inputLabel="Your work email address"
  inputType="email"
  placeholder="me@github.com"
/>
```

## Component props

### CTAForm

| name            | type                                                                                            | default     | required | description                                  |
| --------------- | ----------------------------------------------------------------------------------------------- | ----------- | -------- | -------------------------------------------- |
| `inputLabel`    | `string`                                                                                        | `undefined` | `true`   | The label used by the input component        |
| `inputType`     | `number`, `text`, `email`, `password`, `search`, `tel`, `url`, `date`, `time`, `datetime-local` | `email`     | `false`  | The type of text input that is created       |
| `placeholder`   | `string`                                                                                        | `undefined` | `false`  | The placeholder text for the input component |
| `buttonLabel`   | `string`                                                                                        | `undefined` | `true`   | The label used by the button component       |
| `checkboxLabel` | `string`                                                                                        | `undefined` | `true`   | The label used by the checkbox component     |
