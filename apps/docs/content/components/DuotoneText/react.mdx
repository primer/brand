## Examples

### Default

```jsx live
<Text>
  <DuotoneText>
    <DuotoneText.Emphasis>GitHub Codespaces</DuotoneText.Emphasis> offers a
    complete dev environment in seconds.
  </DuotoneText>
</Text>
```

### With Bento

```jsx live
<Bento>
  <Bento.Item rowSpan={5} flow="column">
    <Bento.Content>
      <Bento.Heading size="3">
        <DuotoneText>
          <DuotoneText.Emphasis>This is my super-sweet</DuotoneText.Emphasis>{' '}
          bento heading
        </DuotoneText>
      </Bento.Heading>
      <Link href="#">Call to action</Link>
    </Bento.Content>
    <Bento.Visual position="50% 100%">
      <img
        alt="placeholder, blank area with an gray background color"
        src="https://via.placeholder.com/600x400/f5f5f5/f5f5f5.png"
      />
    </Bento.Visual>
  </Bento.Item>
</Bento>
```

```jsx live

  </RiverBreakout.A11yHeading>

    <img
      src="https://via.placeholder.com/600x400/f5f5f5/f5f5f5.png"
      alt="placeholder, blank area with an off-white background color"
    />
  </RiverBreakout.Visual>
  <RiverBreakout.Content>
    <Text>
      <DuotoneText>
        <DuotoneText.Emphasis>Lorem ipsum dolor sit amet,</DuotoneText.Emphasis>{' '}
        consectetur adipiscing elit. In sapien sit ullamcorper id. Aliquam

```
