```js
import {Section} from '@primer/react-brand'
```

## Examples

### Default

The default section wraps content in a semantic section element and provides default paddings for the content.

```jsx live
<Section>
  <SectionIntro>
    <SectionIntro.Heading>Section</SectionIntro.Heading>
    <SectionIntro.Description>
      This content is wrapped in a Section component.
    </SectionIntro.Description>
  </SectionIntro>
</Section>
```

### With custom padding

```jsx live
<>
  <Section paddingBlockStart="none" paddingBlockEnd="spacious">
    <SectionIntro>
      <SectionIntro.Heading>Section</SectionIntro.Heading>
      <SectionIntro.Description>
        This section has a custom padding at the top and bottom.
      </SectionIntro.Description>
    </SectionIntro>
  </Section>
</>
```

### With background color

```jsx live
  <Section backgroundColor="var(--brand-color-canvas-inset)">

```

### With background image

```jsx live
<>
  <Section
    backgroundImageSrc="https://github.com/user-attachments/assets/068634a3-926d-49b8-b949-0b1e2a204651"
    backgroundImageSize="cover"
    backgroundImagePosition="top center"

      <Grid>
        <Grid.Column>
          <SectionIntro>
            <SectionIntro.Heading>Section</SectionIntro.Heading>
            <SectionIntro.Description>
              This section has a custom background image.
            </SectionIntro.Description>
          </SectionIntro>
        </Grid.Column>
      </Grid>
    </ThemeProvider>
  </Section>
</>
```

### Rounded

```jsx live
<>
  <ThemeProvider colorMode="dark">
    <Section backgroundColor="var(--brand-color-canvas-default)"></Section>
  </ThemeProvider>
  <Section backgroundColor="var(--brand-color-canvas-default)" rounded>
    <Grid>
      <Grid.Column>
        <SectionIntro>
          <SectionIntro.Heading>Section</SectionIntro.Heading>
          <SectionIntro.Description>
            This section has rounded top corners that overlap the previous
            section.
          </SectionIntro.Description>
        </SectionIntro>
      </Grid.Column>
    </Grid>
  </Section>
</>
```

## Component props

### Section

| name                      | type                                                                     | default    | required | description                                                                                                                                                   |
| ------------------------- | ------------------------------------------------------------------------ | ---------- | -------- | ------------------------------------------------------------------------------------------------------------------------------------------------------------- |
| `paddingBlockStart`       | <PropTableValues values={[...SectionPaddingVariants]} commaSeparated  /> | `'normal'` | `false`  | Control the padding at the top of the section.                                                                                                                |
| `paddingBlockEnd`         | <PropTableValues values={[...SectionPaddingVariants]} commaSeparated  /> | `'normal'` | `false`  | Control the padding at the bottom of the section.                                                                                                             |
| `backgroundColor`         | `string` \| `string[]`                                                   |            | `false`  | Add a background color.                                                                                                                                       |
| `backgroundImageSrc`      | `string` \| `string[]`                                                   |            | `false`  | Add one or more background images                                                                                                                             |
| `backgroundImageSize`     | `string` \| `string[]`                                                   | `'cover'`  | `false`  | Control the size of the background image(s). This accepts all CSS `background-size` supported values, including keywords like `contain`.                      |
| `backgroundImagePosition` | `string` \| `string[]`                                                   | `'50%`     | `false`  | Control the position of the background image(s). This accepts all CSS `background-position` supported values, including keywords like `top` and edge offsets. |
